#! /usr/bin/bash

current_location=$(pwd)

sudo apt-get update -y
sudo apt-get upgrade -y

sudo apt-get install man ifupdown iptables apache2 libapache2-mod-wsgi-py3 dhcpcd5 dnsmasq hostapd -y
sudo systemctl stop dnsmasq
sudo systemctl stop hostapd

# make backup folder with path to it
mkdir backup
backup_location=$(pwd)/backup/

#copy default hostapd file
cp /etc/default/hostapd $backup_location

#edit the DEAMON parameters
sudo sed -i 's,#DAEMON_CONF="",DAEMON_CONF="/etc/hostapd/hostapd.conf",' /etc/default/hostapd
sudo sed -i 's,#DAEMON_OPTS="",DAEMON_OPTS="-B",' /etc/default/hostapd
